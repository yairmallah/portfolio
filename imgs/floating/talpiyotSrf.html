14   	def radial_heat_calc(center: geometry_analisys.Point, p: geometry_analisys.Point, para=0.5, max_rad=0) -> float:
15   		distance = center.dist(p)
16   		if max_rad != 0 and distance >= max_rad:
17   			return 0
18   		return round(-distance*para) % 256

20   	@staticmethod
21   	def bump_func_1(center: geometry_analisys.Point, p: geometry_analisys.Point, para=20, max_rad=0) -> float:
22   		distance = center.dist(p)
23   		if max_rad != 0 and distance >= max_rad:
24   			return 0
25   		return 256 / ((np.sqrt(2.0 * np.pi) * para) * np.exp(-np.power(distance / para, 2.0) / 2))

27   	@staticmethod
28   	def bump_func_2(center: geometry_analisys.Point, p: geometry_analisys.Point, para=20, max_rad=0) -> float:
29   		distance = center.dist(p)
30   		if max_rad != 0 and distance >= max_rad:
31   			return 0
32   		return -distance*para

232  	def view_points_series(self, rang: list, point: (geometry_analisys.Point, list), dist_f=0, pozitivieszed=True):
233  		""" 
234          ripples series from wighted max points 
235          normalize each than apply waves 
236          :param rang: series of radius's 
237          :param point: center pts, list of pts ant wight (scalar) 
238          :param dist_f: function, preferably from static opts 
239          :param pozitivieszed: cuts negative values 
240          :return: circular ripples 
241          """
242  		# BASE: 50 IMG, 5 PTS
243  		# 0: ?
244  		# 1: ?
245  		# 2: 62S
246  		# 3: 37S
247  		dist_fs = {
248  			0: lambda c, p1, v: c.dist(p1),
249  			1: lambda c, p1, v: self.radial_heat_calc(center=c, p=p1),
250  			2: lambda c, p1, v: self.bump_func_1(center=c, p=p1),
251  			3: lambda c, p1, v: self.bump_func_2(center=c, p=p1),
252  			4: lambda c, p1, v: self.bump_func_3(center=c, p=p1, para=v),
253  		}
254  
255  		dist_f = dist_fs[dist_f]
256  		img = ((self.__base_img.copy() * self.BG_BASE_FACTOR).astype(dtype=float))*0
257  
258  		for i in range(len(img)):
259  			if (i * 100) // len(img) > ((i - 1) * 100) // len(img):
260  				print(str((i/len(img)) * 100) + "%")
261  			for j in range(len(img[i])):
262  				p2 = geometry_analisys.Point(i, j)
263  				if type(point) is list:
264  					vals = [dist_f(p, p2, v) for p, v in point]
265  					img[i, j, 1] = max(vals)
266  				else:
267  					img[i, j, 1] = dist_f(point, p2, 1)
268  		imgs = []
269  		img[:, :, 1] -= np.max(img[:, :, 1])
270  		i = 1
271  		for mec in rang:
272  			img2 = np.zeros(img.shape)
273  			mec_add = np.ones(img.shape[:-1])
274  			mec_add *= mec
275  			img2[:, :, 1] = (img[:, :, 1] + mec_add)  # * (255 / np.max(img[:, :, 1] / mec))
276  			if pozitivieszed:
277  				img2[:, :, 1] = (img2[:, :, 1] + np.absolute(img2[:, :, 1])) * 0.5
278  			mol = (i/(len(rang)*0.4))**1.5
279  			if mol > 1:
280  				mol = mol ** 2
281  				if mol > 3:
282  					mol += (mol - 3) ** 2
283  			img2 *= mol
284  			img2[:, :, 1] = img2[:, :, 1].astype(dtype=np.uint8)
285  			imgs.append(img2.astype(dtype=np.uint8))
286  			i += 1
287  		print("end view_points_series, generated %d images" % len(imgs))
288  		return imgs

290  	def img_maximas(self, img: np.ndarray, threshold=150, neighborhood_size=25) -> list:
291  		if img is None:
292  			img = self.__base_img
293  
294  		data_max = ndimage.maximum_filter(img, neighborhood_size)
295  		maxima = (img == data_max)
296  		data_min = ndimage.minimum_filter(img, neighborhood_size)
297  		diff = ((data_max - data_min) > threshold)
298  		maxima[diff == 0] = 0
299  
300  		labeled, num_objects = ndimage.label(maxima)
301  		slices = ndimage.find_objects(labeled)
302  
303  		x, y, v = [], [], []
304  		for dy, dx, _ in slices:
305  			x_center = (dx.start + dx.stop - 1) // 2
306  			x.append(x_center)
307  			y_center = (dy.start + dy.stop - 1) // 2
308  			y.append(y_center)
309  			v_sum = 0
310  			for i in range(-1, 2):
311  				for j in range(-1, 2):
312  					v_sum += max([abs((float(img[y_center + i, x_center + j, q]) - float(img[y_center, x_center, q]))) for q in range(img.shape[2])])
313  			v.append(v_sum/8)
314  		mxv = max(v)
315  		mnv = min(v)
316  		if mxv != 0:
317  			for i in range(len(v)):
318  				v[i] = (v[i] - mnv) / mxv + 0.5
319  		# cv2.circle(img, (x_center, y_center), 3, [0, 0, 255], 3)
320  		print("max's count = " + str(len(x)))
321  		# todo: decide about images: to draw or not to draw
322  		return [(geometry_analisys.Point(x[index], y[index]), v[index]) for index in range(len(x))]
